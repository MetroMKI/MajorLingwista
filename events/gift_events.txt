namespace = gift_events

# Un don pour la Major ! (100 bits)
# Inspiré par l'événement "Gift to the State" (https://eu4.paradoxwikis.com/Prestige_events)
country_event = {
	id = gift_events.1
	title = "gift_event.1.title"
	desc = "gift_event.1.desc"
	picture = CORRUPTION_eventPicture
	
	is_triggered_only = yes

	immediate = {
		every_owned_province = {
			# if base_tax < min_base_tax
			limit = {
				NOT = { higher_development_than = event_target:least_developed_province }
			}
			save_event_target_as = least_developed_province
		}
	}


	option = {
		name = "gift_event.a"
		add_years_of_income = 0.4
	}
	option = {
		name = "gift_event.b"
		add_stab_or_prestige = {THRESHOLD=3 STAB=1 PRESTIGE=5}
	}
	option = {
		name = "gift_event.c.1"
		event_target:least_developed_province = {
			add_base_tax = 1
		}
		
	}
}
