namespace = gift_events

# Inspiré par l'événement "Gift to the State" (https://eu4.paradoxwikis.com/Prestige_events)

# Un don pour la Major (Tier 1) !
country_event = {
	id = gift_events.1
	title = "gift_event.1.title"
	desc = "gift_event.1.desc"
	picture = CORRUPTION_eventPicture
	
	is_triggered_only = yes

	immediate = {
		every_owned_province = {
			# if base_tax < min_base_tax
			limit = {
				NOT = { higher_development_than = event_target:least_dev }
			}
			save_event_target_as = least_dev
		}
	}

	option = {
		name = "gift_event.income"
		add_years_of_income = 0.4
	}
	option = {
		name = "gift_event.stab_or_prestige"
		add_stab_or_prestige = {THRESHOLD=3 STAB=1 PRESTIGE=5}
	}
	option = {
		name = "gift_event.least_dev_tax"

		event_target:least_dev = {
			add_base_tax = 1
		}
	}
	option = {
		name = "gift_event.least_dev_prod"

		event_target:least_dev = {
			add_base_production = 1
		}
	}
	option = {
		name = "gift_event.least_dev_mil"

		event_target:least_dev = {
			add_base_manpower = 1
		}
	}
}

# Un généreux don pour la Major (Tier 2) !
country_event = {
	id = gift_events.2
	title = "gift_event.2.title"
	desc = "gift_event.2.desc"
	picture = CORRUPTION_eventPicture
	
	is_triggered_only = yes

	immediate = {
		random_owned_province = {
			save_event_target_as = most_developed_province
		}
		every_owned_province = {
			# if base_tax > min_base_tax
			limit = {   
					higher_development_than = event_target:most_developed_province
			}
			save_event_target_as = most_developed_province		
		}
	}

	option = {
		name = "gift_event.a"
		add_years_of_income = 2
	}
	option = {
		name = "gift_event.b"
		add_stab_or_prestige = {THRESHOLD=2 STAB=2 PRESTIGE=25}
	}
	option = {
		name = "gift_event.c.2"
		event_target:most_developed_province = {
			add_base_tax = 1
		}
	}
}
