# Un don pour la Major ! (100 bits)
# Inspiré par l'événement "Gift to the State" (https://eu4.paradoxwikis.com/Prestige_events)
country_event = {
	id = gift_events.1
	title = "gift_event.1.title"
	desc = "gift_event.1.desc"
	picture = CORRUPTION_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			random_owned_province = {
				save_event_target_as = to_develop_province
			}
		}
	}

	option = {
		name = "gift_event.1.a"
		add_years_of_income = 0.5
	}
	option = {
		name = "gift_event.1.b"
		add_prestige = 5
	}
	option = {
		name = "gift_event.1.c"
		event_target:to_develop_province = {
			province_event = {
				id = gift_events.4
			}
		}
		
	}
}
