country_event = {
	id = majorie_events.1
	title = "majorie_event.1t"
	desc = "majorie_event.1d"
	picture = ANGRY_MOB_eventPicture

	fire_only_once = yes

	trigger = {
		ai = no
		NOT = { has_reform = illuminatos_council_reform }
		NOT = { tag = LGW }
	}
	
	mean_time_to_happen = {
		days = 0

		modifier = {
			factor = 1000
			ai = no
		}
	}

	option = {
		name = "majorie_event.1t"
		add_permanent_claim = 192
		add_permanent_claim = 4384
		if = {
			limit = { primitives = yes }
			set_primitive = no
		}
		if = {
			limit = {
				NOT = { government = republic }
			}
			change_government = republic
		}
		country_event = {
			id = majorie_events.2
			days = 0
		}
	}
}
