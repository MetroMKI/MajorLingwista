

#Reduce Inflation
estate_natifs_restore_currency = {
	can_select = {
		inflation = 10
		NOT = {
			inflation = 20
		}
		NOT = { has_country_modifier = boom_economy }
	}
	selection_weight = {
		factor = 5
	}
	fail_if = {
		inflation = 20
	}
	task_requirements = {
		NOT = { inflation = 5 }
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_natifs
			loyalty = 10
		}
		add_country_modifier = {
			name = boom_economy
			duration = 5475
		}
		add_inflation = -2
    }
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_natifs
			desc = estate_lingwista_echec
			loyalty = -5
			duration = 7300
        }
    }
}

#Recover Abysmal Inflation
estate_natifs_recovery_from_hyperinflation = {
	can_select = {
		inflation = 20
	}
	selection_weight = {
		factor = 5
	}
	task_requirements = {
		NOT = { inflation = 10 }
	}
    task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_natifs
			loyalty = 10
		}
		add_stability_or_adm_power = yes
		add_inflation = -5
    }
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_natifs
			desc = estate_lingwista_echec
			loyalty = -5
			duration = 7300
        }
    }
}


#Get a treasury
estate_natifs_get_cash = {
	can_select = {
		NOT = { years_of_income = 0.5 }
		NOT = { num_of_loans = 1 }
		is_at_war = no
	}
	selection_weight = {
		factor = 1
	}
	task_requirements = {
		years_of_income = 2.5
		NOT = { num_of_loans = 1 }
	}
	fail_if = {
		num_of_loans = 1
	}
	task_completed_effect = {
		add_estate_loyalty = {
			estate = estate_natifs
			loyalty = 15
		}
	}
    failing_effect = {
        add_estate_loyalty_modifier = {
            estate = estate_natifs
			desc = estate_lingwista_echec
			loyalty = -5
			duration = 7300
        }
    }
}
